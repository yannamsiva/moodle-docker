# Dockerfile

FROM php:8.1-apache

# Install dependencies
RUN apt-get update && apt-get install -y \
    git zip unzip libpng-dev libjpeg-dev libfreetype6-dev \
    libxml2-dev libzip-dev libonig-dev libicu-dev \
    && docker-php-ext-install mysqli pdo pdo_mysql intl zip xml gd \
    && docker-php-ext-enable mysqli

# Enable Apache rewrite module
RUN a2enmod rewrite

# Set working directory
WORKDIR /var/www/html

# Copy Moodle source code (copied via volume or local build)
COPY ./moodle /var/www/html

# Permissions
RUN chown -R www-data:www-data /var/www/html \
    && chmod -R 755 /var/www/html
